const boxW = 110, boxH = 30, size = 5000; let svg, g, root, allMembers = []; // Unified Search Function function syncSearch(val) { document.getElementById('treeSearch').value = val; document.getElementById('dirSearch').value = val; renderDirectory(val); applyTreeFilter(val); } /** * FIXED: Tree Filtering Logic * Hides all nodes that don't match the Name or Location substring */ function applyTreeFilter(val) { if (!val || val.trim() === "") { // Reset to default collapsed state if search is empty root.descendants().forEach(d => { if (d.depth > 0 && d.children) { d._children = d.children; d.children = null; } }); renderTree(root); return; } const term = val.toLowerCase(); // 1. Identify all matching nodes (Name or Location) const matches = root.descendants().filter(d => { const nameMatch = d.data.name && d.data.name.toLowerCase().includes(term); const locMatch = d.data.location && d.data.location.toLowerCase().includes(term); return nameMatch || locMatch; }); // 2. Create a Set of IDs that MUST stay visible (Matches + their Ancestors) const visibleIds = new Set(); matches.forEach(d => { d.ancestors().forEach(anc => visibleIds.add(anc.data.id)); }); // 3. Surgical Toggle: If ID not in visibleIds, it gets hidden root.descendants().forEach(d => { if (visibleIds.has(d.data.id)) { // This is part of the search path - Expand it if (d._children) { d.children = d._children; d._children = null; } } else { // Not part of search path - Hide it if (d.children) { d._children = d.children; d.children = null; } } }); renderTree(root); // Auto-center if we have matches if (matches.length > 0) { centerNode(matches[0]); } } async function init() { try { const t = new Date().getTime(); const [membersData, linksData] = await Promise.all([ d3.csv(`members.csv?v=${t}`), d3.csv(`links.csv?v=${t}`) ]); allMembers = membersData; const m = new Map(); membersData.forEach(d => m.set(d.id, d)); const lineages = linksData.filter(l => l.type === 'lineage'); const marriages = linksData.filter(l => l.type === 'marriage'); function build(id) { const u = m.get(id); if (!u) return null; const mar = marriages.find(x => x.parent === id || x.child === id); const spouseId = mar ? (mar.parent === id ? mar.child : mar.parent) : null; const spouse = spouseId ? m.get(spouseId) : null; const k = mar ? mar.parent + '+' + mar.child : id; const alt = mar ? mar.child + '+' + mar.parent : id; const ch = lineages.filter(r => r.parent === k || r.parent === alt || r.parent === id) .map(r => build(r.child)).filter(x => x); return { ...u, spouse, children: ch.length ? ch : null, _direct: ch.length, _total: ch.reduce((a, c) => a + 1 + (c._total || 0), 0) }; } root = d3.hierarchy(build('DH')); root.x0 = size / 2; root.y0 = 0; // Initial state: Collapse all except root root.descendants().forEach(d => { if (d.depth > 0 && d.children) { d._children = d.children; d.children = null; } }); svg = d3.select("#canvas").append("svg").attr("width", size).attr("height", size); g = svg.append("g").attr("transform", `translate(${size / 2}, 100)`); renderTree(root); centerNode(root); } catch (err) { console.error("Data Load Error:", err); } } function renderTree(source) { const tree = d3.tree().nodeSize([boxW * 2.8, 140]); tree(root); const nodes = root.descendants(); const links = root.links(); // Transitions and drawing logic const link = g.selectAll(".link").data(links, d => d.target.data.id); link.exit().remove(); link.enter().append("path").attr("class", "link").merge(link) .transition().duration(500).attr("d", d3.linkVertical().x(d => d.x).y(d => d.y)); const node = g.selectAll(".node").data(nodes, d => d.data.id); node.exit().remove(); const nE = node.enter().append("g").attr("class", "node") .attr("transform", d => `translate(${source.x0 || 0},${source.y0 || 0})`); node.merge(nE).transition().duration(500).attr("transform", d => `translate(${d.x},${d.y})`); nE.each(function (d) { const el = d3.select(this); if (d.data.spouse) { el.append("rect").attr("class", "marriage-rect").attr("x", -boxW - 10).attr("y", -boxH / 2 - 5).attr("width", boxW * 2 + 20).attr("height", boxH + 15); drawBox(el, d.data, -boxW / 2 - 5); drawBox(el, d.data.spouse, boxW / 2 + 5); } else drawBox(el, d.data, 0); if (d.children || d._children) { const ui = el.append("g").attr("transform", `translate(0, ${boxH / 2 + 12})`).on("click", (e, nD) => { e.stopPropagation(); nD.children ? (nD._children = nD.children, nD.children = null) : (nD.children = nD._children, nD._children = null); renderTree(nD); }); ui.append("rect").attr("x", -20).attr("y", -7).attr("width", 40).attr("height", 14).attr("fill", "#2c3e50").attr("rx", 7); ui.append("text").attr("y", 4).attr("text-anchor", "middle").style("fill", "white").style("font-size", "10px").text(`${d.data._direct}|${d.data._total}`); ui.append("circle").attr("cy", 18).attr("r", 7).attr("fill", "white").attr("stroke", "#2c3e50"); ui.append("text").attr("y", 22).attr("text-anchor", "middle").style("fill", "#2c3e50").style("font-size", "12px").text(d.children ? "-" : "+"); } }); nodes.forEach(d => { d.x0 = d.x; d.y0 = d.y; }); } function drawBox(c, d, o) { const b = c.append("g").on("click", (e) => { e.stopPropagation(); showCard(d); }); b.append("rect").attr("x", o - boxW / 2).attr("y", -boxH / 2).attr("width", boxW).attr("height", boxH).attr("fill", d.gender === 'M' ? ID: ${m.id} | ${m.location || 'No Location'}
`; d.onclick = () => showCard(m); l.appendChild(d); }); } // Modal and utility functions remain unchanged...
